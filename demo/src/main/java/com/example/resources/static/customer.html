function login(){ let acc_num = document.getElementById("acc_num").value; let pin = document.getElementById("pin").value; fetch("https://code-bank-management.railway.app/login", { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded" }, body: new URLSearchParams({ acc_num: acc_num, pin: pin }) }) .then(res => res.text()) .then(data => { if(data.includes("Welcome")){ localStorage.setItem("acc_num", acc_num); window.location.href = "dashboard.html"; } else { document.getElementById("result").innerText = data; } }) .catch(err => { document.getElementById("result").innerText = "Server Error: " + err; }); }
